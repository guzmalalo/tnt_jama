   #define MATH_UTILS_H
   
   
   #include <cmath> // std::abs(), std::sqrt()
   #include <algorithm> // std::min(), std=::max()
   
   namespace TNT
   {
   
   
   template <class T>
   T hypot(const T &a, const T &b)
   {
       if (a== 0)
           return std::abs(b);
       else
       {
           T c = b/a;
           return std::abs(a) * std::sqrt(1 + c*c);
       }
   }
   } /* TNT namespace */
   
   
   
   #endif
   /* MATH_UTILS_H */
